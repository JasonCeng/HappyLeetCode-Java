import java.util.Arrays;
import java.util.Scanner;

public class Main {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);
        int n = sc.nextInt();
        long[] array = new long[n];
        for(int i = 0; i < n; i++) {
            array[i] = sc.nextLong();
        }
        maxMul(array, n);
    }
    static void maxMul(long[] arr, int len) {
      long max = 1, sec_max = 1, thir_max = 1, min = 1, sec_min = 1;
      for(int i = 0; i < len; i++) {
            if(arr[i] > max) {
                thir_max = sec_max;
                sec_max = max;
                max = arr[i];
            }
            if(arr[i] > sec_max && arr[i] < thir_max) {
                thir_max = sec_max;
                sec_max = arr[i];
            }
            if(arr[i] > thir_max) {
                thir_max = arr[i];
            }
            if(arr[i] < min) {
                sec_min = min;
                min = arr[i];
            }
            if(arr[i] < sec_min && arr[i] > min) {
                sec_min = arr[i];
            }
        }
        long maxResult = Math.max(max * sec_max * thir_max,max * min * sec_min);
        System.out.println(maxResult);
    }
}
